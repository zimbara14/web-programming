Программа для нахождения факториала любого числа

      
def findFactorial(n):
  if n < 0:
    return ("Sorry, factorial does not exist for negative numbers")
  elif n == 0:
    return 1
  else:
    factorial = 1
    for i in range(1, n + 1):
      factorial *= i
    return factorial
